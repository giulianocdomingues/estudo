# comandos git # Ferramenta de visualizacao de funcionamento do git: https://git-school.github.io/visualizing-git/ git add . git status git log git log --oneline # traz todas as alteracoes em uma linha git commit -m "teste" git remote # Lista os servidores remotos git remote add NOME_DO_REPO /caminho/do/servidor/remoteo # Adiciona servidor remoto com nome especifico git remote remove NOME_DO_REPO git remote rename origin NOME_DO_REPO git push NOME_DO_REPO NOME_DA_BRANCH # Envia arquivos para repositorio remoto com branch especifica git pull NOME_DO_REPO NOME_DA_BRANCH # Baixa arquivos para repositorio remoto com branch especifica git branch # Lista branches ativas git branch BRANCH_NAME # Adiciona uma branch com nome especifico git checkout BRANCH_NAME # Altera branch ativa master para criada git rebase BRANCH # Carrega alteracoes de branches na master git checkout -- # Volta o arquivo anterior sem as alteracoes. git checkout hash_commit # Volta o arquivo para a modificação criada no momento do commit git revert nome_do_arquivo # Volta estado do arquivo antes do add. Realizado isto é necessario o comando anterior. Em caso de commit, pode-se pegar o hash da ultima versao com o comando git log e aplicar o git rever HASH git stash # Salva o arquivo em local temporario para depois aplica-lo. git stash list # lista as informacoes de stashs salvos git stash pop # aplica stash no arquivo com alterações git diff hash1 hash2 # Apresenta as alteracoes realizadas entre os comits.
